async function attemptLogin() { const passkey = document.getElementById('passkey-input').value; const status = document.getElementById('login-status'); const webAppUrl = "URL_WEB_APP_ABANG_YANG_BARU"; // Tempel link /exec di sini status.innerText = "Memverifikasi..."; try { const response = await fetch(webAppUrl, { method: "POST", body: JSON.stringify({ action: "login", passkey: passkey }) }); const res = await response.json(); if (res.success) { // Simpan nama admin di memori browser sebentar localStorage.setItem("admin_name", res.admin_name); // Pindah ke halaman dashboard window.location.href = "dashboard.html"; } else { status.innerText = "Passkey Salah, Bre!"; } } catch (err) { status.innerText = "Koneksi Gagal!"; console.error(err); } }
